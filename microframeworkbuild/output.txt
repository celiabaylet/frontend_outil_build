import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';
import config from './plugo.config.js';

// üîß Configuration du chemin courant (√©quivalent de __dirname en ESM)
const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

// 1Ô∏è‚É£ Chemins
const cssPath = path.join(__dirname, 'css');
const distPath = path.join(__dirname, 'dist');

// 2Ô∏è‚É£ Cr√©e le dossier dist s‚Äôil n‚Äôexiste pas
if (!fs.existsSync(distPath)) {
  fs.mkdirSync(distPath);
}

// 3Ô∏è‚É£ G√©n√®re les variables CSS depuis la config
let cssOutput = `/* ======================================
   PLUGO CSS BUILD
   G√©n√©r√© automatiquement le ${new Date().toLocaleString()}
====================================== */

:root {
  --color-primary: ${config.theme.colors.primary};
  --color-success: ${config.theme.colors.success};
  --color-warning: ${config.theme.colors.warning};
  --color-danger: ${config.theme.colors.danger};

  --font-main: ${config.theme.typography.main};
  --font-headlines: ${config.theme.typography.headlines};

  --container-width: ${config.theme.layout.container};
  --base-unit: ${config.theme.spacing.baseUnit};
  --transition-duration: ${config.theme.transition.duration};
  --transition-type: ${config.theme.transition.type};
}

`;

// 4Ô∏è‚É£ Fonction pour fusionner les fichiers CSS d‚Äôun dossier
function mergeCssFiles(folderPath) {
  let combined = '';
  const files = fs.readdirSync(folderPath);
  for (const file of files) {
    if (file.endsWith('.css')) {
      const filePath = path.join(folderPath, file);
      const content = fs.readFileSync(filePath, 'utf8');
      combined += `\n/* ===== ${file} ===== */\n${content}\n`;
    }
  }
  return combined;
}

// 5Ô∏è‚É£ Fusion de toutes les parties
cssOutput += mergeCssFiles(path.join(cssPath, 'base'));
cssOutput += mergeCssFiles(path.join(cssPath, 'layout'));

// Components choisis depuis la config
for (const comp of config.components) {
  const compPath = path.join(cssPath, 'components', `_${comp}.css`);
  if (fs.existsSync(compPath)) {
    cssOutput += `\n/* === Component: ${comp} === */\n`;
    cssOutput += fs.readFileSync(compPath, 'utf8');
  }
}

// Utilities choisies depuis la config
for (const util of config.utilities) {
  const utilPath = path.join(cssPath, 'utilities', `_${util}.css`);
  if (fs.existsSync(utilPath)) {
    cssOutput += `\n/* === Utility: ${util} === */\n`;
    cssOutput += fs.readFileSync(utilPath, 'utf8');
  }
}

// Mode sombre (si activ√©)
if (config.darkMode) {
  cssOutput += `
@media (prefers-color-scheme: dark) {
  body {
    background-color: #111;
    color: #eee;
  }
}
`;
}

// 6Ô∏è‚É£ √âcriture du fichier final
fs.writeFileSync(path.join(distPath, 'plugo.css'), cssOutput, 'utf8');
console.log('‚úÖ Fichier dist/plugo.css g√©n√©r√© avec succ√®s !');
